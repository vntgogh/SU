#include <stdio.h>
#include <stdlib.h>
#include <math.h>

double somme(double n){
    if(n==0){
        return 1;
    }else{
        return n * somme(n-1);
    }
}

double sommef(double nb){
    if (nb==0){
        return 1;
    }else{
        return somme(nb)+sommef(nb-1);
    }
}

void frac(int k){
    int y;
    for(int x=0; x<=k; x++){
        y=k*x/(x-k);
        if(y>0){
            if((1/k)==((1/x)+(1/y))){
                printf("1/%d = 1/%d + 1/%d\n", k,x,y);
            }
        }
    }
}

int main(){
    printf("%.0f\n", sommef(10));
    printf("%.0f\n", sommef(13));
    printf("%.0f\n", sommef(20));

    frac(5);
    return 0;
}
